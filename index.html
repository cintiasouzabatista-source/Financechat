function createCard(text) { const container = document.getElementById('chat-container'); const card = document.createElement('div'); // Lógica simples de extração (Simulando a IA) const valorTotal = parseFloat(text.match(/\d+/)) || 0; const parcelasMatch = text.match(/(\d+)\s*x/) || text.match(/(\d+)\s*vezes/); const numParcelas = parcelasMatch ? parseInt(parcelasMatch[1]) : 1; const valorParcela = (valorTotal / numParcelas).toFixed(2); const conta = text.toLowerCase().includes('inter') ? 'Banco Inter' : 'Nubank'; // Exemplo const categoria = text.toLowerCase().includes('uber') ? 'Transporte' : 'Compras'; const isParcelado = numParcelas > 1; card.className = `card-mov ${isParcelado ? 'parcelado' : 'despesa'}`; card.innerHTML = `
${categoria} ${conta}

Valor Total: R$ ${valorTotal.toFixed(2)}

${isParcelado ? `

Parcelamento: ${numParcelas}x de R$ ${valorParcela}

⚠️ Lançado na fatura dos próximos meses

` : `

✅ À vista: Debitado do saldo agora

`}

Data: ${new Date().toLocaleDateString('pt-BR')}

`; container.appendChild(card); container.scrollTop = container.scrollHeight; // Inicia o timer de 8s (mesma lógica anterior) }
