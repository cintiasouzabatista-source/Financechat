// INTEGRAÇÃO: CADASTRO -> SALDOS function salvarImediato() { const desc = document.getElementById('rev-desc').value; const valor = parseFloat(document.getElementById('rev-valor').value.replace(',', '.')); const cat = document.getElementById('rev-cat').value; const fixa = document.getElementById('rev-fixa').value === 'sim'; // 1. Atualiza Saldo Geral saldoInicial -= valor; saldoEl.innerText = `R$ ${saldoInicial.toLocaleString('pt-BR', { minimumFractionDigits: 2 })}`; // 2. Alimenta os dados do Gráfico if (dadosCategorias[cat]) { dadosCategorias[cat] += valor; } else { dadosCategorias[cat] = valor; } // 3. Se for Fixa, projeta para os próximos 11 meses internamente if (fixa) { console.log(`Programando recorrência: ${desc} repetido por 12 meses.`); } // 4. Atualiza a UI sem recarregar a página adicionarLista(desc, valor, fixa); atualizarInterfaceGrafico(); fechar(); } // Lógica de Edição em Lote ao clicar em um item recorrente function confirmarExclusaoCompleta(btn, ehRecorrente) { if (ehRecorrente) { abrirModalLote("ID_DO_LANCAMENTO"); // Abre o modal que criamos antes } else { if(confirm("Excluir este lançamento?")) btn.closest('.movimentacao-card').remove(); } }
